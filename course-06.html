🏗️面向对象编程 - 类与对象

ES6引入了class语法，使JavaScript的面向对象编程更加直观和易于理解。

class Car { // 构造函数 constructor(brand, model, year) { this.brand = brand; // 品牌 this.model = model; // 型号 this.year = year; // 年份 this.isRunning = false; // 运行状态 this.speed = 0; // 当前速度 } // 启动方法 start() { this.isRunning = true; return `🚗 ${this.brand} ${this.model} 启动成功！`; } // 停止方法 stop() { this.isRunning = false; this.speed = 0; return `🛑 ${this.brand} ${this.model} 已停止`; } // 加速方法 accelerate(increment = 10) { if (this.isRunning) { this.speed += increment; return `⚡ 加速中... 当前速度: ${this.speed} km/h`; } return "❌ 请先启动汽车！"; } // 获取信息 getInfo() { return { brand: this.brand, model: this.model, year: this.year, status: this.isRunning ? '运行中' : '停止', speed: this.speed }; } }

🔗继承 - 扩展类的功能

通过继承，我们可以创建更专业化的类，复用父类的功能并添加新特性。

class ElectricCar extends Car { constructor(brand, model, year, batteryCapacity) { super(brand, model, year); // 调用父类构造函数 this.batteryCapacity = batteryCapacity; // 电池容量(kWh) this.batteryLevel = 100; // 电量百分比 this.chargingStatus = false; // 充电状态 } // 充电方法 charge() { this.chargingStatus = true; const chargeInterval = setInterval(() => { if (this.batteryLevel < 100) { this.batteryLevel += 5; this.updateChargeDisplay(); } else { this.chargingStatus = false; clearInterval(chargeInterval); this.updateChargeDisplay("充电完成！"); } }, 200); return "🔌 开始充电..."; } // 行驶方法 drive(distance) { if (this.batteryLevel <= 0) { return "🔋 电量不足，请充电！"; } if (!this.isRunning) { return "❌ 请先启动汽车！"; } const consumption = distance * 0.2; // 每公里消耗0.2%电量 this.batteryLevel = Math.max(0, this.batteryLevel - consumption); return `🛣️ 行驶了 ${distance} 公里，剩余电量: ${this.batteryLevel.toFixed(1)}%`; } // 更新充电显示 updateChargeDisplay(message = null) { const output = document.getElementById('electricCarOutput'); if (message) { output.textContent = message; } else { output.textContent = `🔌 充电中... 当前电量: ${this.batteryLevel}%`; } this.updateBatteryProgress(); } // 更新电池进度条 updateBatteryProgress() { const progress = document.getElementById('batteryProgress'); if (progress) { progress.style.width = this.batteryLevel + '%'; progress.style.background = this.batteryLevel > 20 ? 'linear-gradient(135deg, #2ecc71, #27ae60)' : 'linear-gradient(135deg, #e74c3c, #c0392b)'; } } // 重写获取信息方法 getInfo() { const baseInfo = super.getInfo(); return { ...baseInfo, batteryCapacity: this.batteryCapacity, batteryLevel: this.batteryLevel.toFixed(1), chargingStatus: this.chargingStatus ? '充电中' : '未充电' }; } }

⏰异步编程 - 回调函数与定时器

JavaScript是单线程语言，但通过异步编程可以处理耗时操作而不阻塞主线程。

回调函数：作为参数传递给其他函数的函数，在特定条件满足时被调用。
// 基础回调函数示例 function processData(data, callback) { console.log('开始处理数据:', data); // 模拟异步操作 setTimeout(() => { const result = data.toUpperCase(); console.log('数据处理完成:', result); // 调用回调函数 if (callback && typeof callback === 'function') { callback(result); } }, 2000); } // 使用回调函数 processData('hello world', (result) => { console.log('回调函数执行，结果是:', result); alert('处理完成: ' + result); }); // 多层回调示例 function delayedMessage(message, delay, callback) { setTimeout(() => { console.log(`⏰ ${delay}ms后: ${message}`); if (callback) callback(); }, delay); }

🎨DOM操作 - 动态网页交互

DOM (Document Object Model) 操作是JavaScript与网页交互的核心技术。

DOM操作核心概念：通过JavaScript动态修改HTML元素的内容、样式和行为。
// 获取元素的多种方式 const elementById = document.getElementById('myId'); const elementsByClass = document.getElementsByClassName('myClass'); const elementsByTag = document.getElementsByTagName('div'); const querySelector = document.querySelector('.my-selector'); const querySelectorAll = document.querySelectorAll('.my-class'); // 修改元素内容 element.textContent = '新的文本内容'; // 纯文本 element.innerHTML = 'HTML内容'; // HTML内容 // 修改元素样式 element.style.color = '#e74c3c'; element.style.fontSize = '20px'; element.style.background = 'linear-gradient(135deg, #667eea, #764ba2)'; // 添加/移除CSS类 element.classList.add('new-class'); element.classList.remove('old-class'); element.classList.toggle('active'); // 事件监听 element.addEventListener('click', function(event) { console.log('元素被点击了!', event); });
